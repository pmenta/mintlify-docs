---
title: "Find Many Tracking"
openapi: "GET /tracking"
api: "GET https://tracking.stage.judit.io/tracking"
description: "Esse endpoint consulta monitoramentos por parâmetros de forma paginada"
---

<ParamField query="page" type="integer" required={false}>
    Define a página dos resultados que você deseja consultar.
</ParamField>

<ParamField query="pageSize" type="integer" required={false}>
    Define o número máximo de resultados que você deseja receber por página;
</ParamField>

<ParamField query="orderField" type="string" required={false}>
    Nome do campo em que se deseja ordenar os buscas;
</ParamField>

<ParamField query="orderDirection" type="string" required={false}>
    Sentido da ordenação: ‘ASC’ ou ‘DESC’;
</ParamField>

<ParamField query="user_id" type="string" required={false}>
    Retorna buscas relacionadas ao usuário;
</ParamField>

<ParamField query="reference_type" type="string" required={false}>
    Retorna monitoramentos do tipo de referência especificado “cpf”, “cnpj”, “oab” ou “lawsuit_id”;
</ParamField>

<ParamField query="reference_id" type="string" required={false}>
    Retorna monitoramentos relacionado ao número do CPF, CNPJ, OAB ou Código CNJ;
</ParamField>

<ParamField query="status" type="string" required={false}>
    Retorna monitoramentos cujo status podem ser 'created', 'updating', 'updated', 'paused' ou 'deleted' ou mais de um de status ['updating', 'paused'];
</ParamField>

<ParamField query="tags" type="string" required={false}>
    Retorna monitoramentos com as tags informadas;
</ParamField>

<ParamField query="recurrence_gte" type="date" required={false}>
    Retorna monitoramentos com a recorrência de atualização maior ou igual a informada;
</ParamField>

<ParamField query="recurrence_lte" type="date" required={false}>
    Retorna monitoramentos com a recorrência de atualização menor ou igual a informada;
</ParamField>

<ParamField query="created_at_gte" type="date" required={false}>
    Retorna buscas com a data de criação maior ou igual a informada;
</ParamField>

<ParamField query="created_at_lte" type="date" required={false}>
    Retorna buscas com a data de criação menor ou igual a informada;
</ParamField>

<ParamField query="updated_at_gte" type="date" required={false}>
    Retorna buscas com a data de atualização maior ou igual a informada;
</ParamField>

<ParamField query="updated_at_lte" type="date" required={false}>
    Retorna buscas com a data de atualização menor ou igual a informada;
</ParamField>

<ParamField query="updating_at_gte" type="date" required={false}>
    Retorna monitoramentos com a data de execução do monitoramento maior ou igual a informada;
</ParamField>

<ParamField query="updating_at_lte" type="date" required={false}>
    Retornam onitoramentos com a data de execução do monitoramento menor ou igual a informada;
</ParamField>

<ParamField query="paused_at_gte" type="date" required={false}>
    Retorna monitoramentos com a data de pause do monitoramento maior ou igual a informada;
</ParamField>

<ParamField query="paused_at_lte" type="date" required={false}>
    Retorna monitoramentos com a data de pause do monitoramento menor ou igual a informada;
</ParamField>

<ParamField query="deleted_at_gte" type="date" required={false}>
    Retorna monitoramentos com a data de deleção maior ou igual a informada;
</ParamField>

<ParamField query="deleted_at_lte" type="date" required={false}>
    Retorna monitoramentos com a data de deleção menor ou igual a informada;
</ParamField>

<RequestExample>
    ```bash Request
    curl -X GET "https://tracking.stage.judit.io/tracking?page=1&pageSize=10"
    -H "api-key: YOUR_API_KEY"
    ```
</RequestExample>
